CREATE TABLE customers(customer_id integer primary key, name text, age integer, city text, street text, zip_code text);
CREATE TABLE employees(employee_id integer primary key, name text, pay float);
CREATE TABLE contracts(employee_id integer, start_date date, end_date date, function text, foreign key(employee_id) references employees(employee_id));
CREATE TABLE drinks(drink_id integer primary key, name text, alcohol_percentage float, volume float, price float);
CREATE TABLE stock(drink_id integer, amount integer, foreign key(drink_id) references drinks(drink_id));
CREATE TABLE strong_drinks(drink_id integer, type text, foreign key(drink_id) references drinks(drink_id));
CREATE TABLE beers(drink_id, type text, IBU integer, foreign key(drink_id) references drinks(drink_id));
CREATE TABLE orders(order_id integer primary key, customer_id integer, drink_id integer, employee_id integer, time date, amount integer,
foreign key(customer_id) references customers(customer_id), foreign key(drink_id) references drinks(drink_id), foreign key(employee_id) references employees(employee_id));
